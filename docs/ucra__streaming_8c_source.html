63/* Simple sine wave generator for testing/demo purposes */
64static void generate_sine_wave(float* buffer, uint32_t frames, uint32_t channels,
65 uint32_t sample_rate, double* phase, float frequency) {
66 const double phase_increment = 2.0 * M_PI * frequency / sample_rate;
67
68 for (uint32_t frame = 0; frame < frames; frame++) {
69 float sample = (float)(sin(*phase) * 0.1); /* Low volume to prevent ear damage */
70
71 for (uint32_t ch = 0; ch < channels; ch++) {
72 buffer[frame * channels + ch] = sample;
73 }
74
75 *phase += phase_increment;
76 if (*phase >= 2.0 * M_PI) {
77 *phase -= 2.0 * M_PI;
78 }
79 }
80}

82/* Render audio based on note segments from the callback */
83static UCRA_Result render_audio_from_notes(UCRA_StreamState* state,
84 const UCRA_RenderConfig* render_config,
85 float* output_buffer,
86 uint32_t frames_to_render) {
87 /* Clear output buffer first */
88 memset(output_buffer, 0, frames_to_render * state->config.channels * sizeof(float));
89
90 /* If no notes provided, generate silence */
91 if (!render_config->notes || render_config->note_count == 0) {
92 return UCRA_SUCCESS;
93 }
94
95 /* Simple implementation: render each note as a sine wave and mix them */
96 for (uint32_t note_idx = 0; note_idx < render_config->note_count; note_idx++) {
97 const UCRA_NoteSegment* note = &render_config->notes[note_idx];
98
99 /* Skip notes that are not active in the current time range */
100 double current_time = (double)state->total_frames_generated / state->config.sample_rate;
101 double block_duration = (double)frames_to_render / state->config.sample_rate;
102
103 if (current_time >= note->start_sec + note->duration_sec ||
104 current_time + block_duration <= note->start_sec) {
105 continue; /* Note is not active in this time range */
106 }
107
108 /* Calculate frequency from MIDI note */
109 float frequency = 440.0f; /* Default A4 */
110 if (note->midi_note >= 0) {
111 frequency = 440.0f * powf(2.0f, (note->midi_note - 69) / 12.0f);
112 }
113
114 /* Generate sine wave for this note */
115 float* temp_buffer = malloc(frames_to_render * state->config.channels * sizeof(float));
116 if (temp_buffer) {
117 double temp_phase = state->phase; /* Use a temporary phase for mixing */
118 generate_sine_wave(temp_buffer, frames_to_render, state->config.channels,
119 state->config.sample_rate, &temp_phase, frequency);
120
121 /* Mix into output buffer with velocity scaling */
122 float volume = note->velocity / 127.0f * 0.3f; /* Scale velocity to reasonable volume */
123 for (uint32_t i = 0; i < frames_to_render * state->config.channels; i++) {
124 output_buffer[i] += temp_buffer[i] * volume;
125 }
126
127 free(temp_buffer);
128 }
129 }
130
131 return UCRA_SUCCESS;
132}

144UCRA_Result ucra_stream_open(UCRA_StreamHandle* out_stream,
145 const UCRA_RenderConfig* config,
146 UCRA_PullPCM callback,
147 void* user_data) {
148 if (!out_stream || !config || !callback) {
149 return UCRA_ERR_INVALID_ARGUMENT;
150 }
151
152 /* Validate configuration */
153 if (config->sample_rate == 0 || config->channels == 0 || config->block_size == 0) {
154 return UCRA_ERR_INVALID_ARGUMENT;
155 }
156
157 /* Allocate stream state */
158 UCRA_StreamState* state = calloc(1, sizeof(UCRA_StreamState));
159 if (!state) {
160 return UCRA_ERR_OUT_OF_MEMORY;
161 }
162
163 /* Copy configuration */
164 memcpy(&state->config, config, sizeof(UCRA_RenderConfig));
165 state->callback = callback;
166 state->user_data = user_data;
167
168 /* Allocate ring buffer - ensure it's at least 4x the block size for good buffering */
169 uint32_t min_buffer_size = config->block_size * 4;
170 state->buffer_size_frames = (min_buffer_size > DEFAULT_BUFFER_SIZE_FRAMES) ?
171 min_buffer_size : DEFAULT_BUFFER_SIZE_FRAMES;
172
173 size_t buffer_size_bytes = state->buffer_size_frames * config->channels * sizeof(float);
174 state->buffer = malloc(buffer_size_bytes);
175 if (!state->buffer) {
176 free(state);
177 return UCRA_ERR_OUT_OF_MEMORY;
178 }
179
180 /* Initialize buffer to silence */
181 memset(state->buffer, 0, buffer_size_bytes);
182
183 /* Initialize ring buffer state */
184 state->write_pos = 0;
185 state->read_pos = 0;
186 state->available_frames = 0;
187
188 /* Initialize audio generation state */
189 state->phase = 0.0;
190 state->total_frames_generated = 0;
191
192 /* Initialize mutex and condition variable */
193 int result = pthread_mutex_init(&state->mutex, NULL);
194 if (result != 0) {
195 free(state->buffer);
196 free(state);
197 return UCRA_ERR_INTERNAL;
198 }
199
200 result = pthread_cond_init(&state->data_available, NULL);
201 if (result != 0) {
202 pthread_mutex_destroy(&state->mutex);
203 free(state->buffer);
204 free(state);
205 return UCRA_ERR_INTERNAL;
206 }
207
208 state->is_initialized = 1;
209 state->is_closed = 0;
210
211 /* Pre-fill buffer with initial data to reduce latency */
212 pthread_mutex_lock(&state->mutex);
213 for (int i = 0; i < 3; i++) { /* Fill 3 blocks initially */
214 UCRA_Result prefill_result = refill_stream_buffer(state);
215 if (prefill_result != UCRA_SUCCESS) {
216 break; /* Stop if callback fails, but don't fail the open */
217 }
218 if (get_available_space_frames(state) < state->config.block_size) {
219 break; /* Buffer is getting full */
220 }
221 }
222 pthread_mutex_unlock(&state->mutex);
223
224 *out_stream = (UCRA_StreamHandle)state;
225 return UCRA_SUCCESS;
226}

228void ucra_stream_close(UCRA_StreamHandle stream) {
229 if (!stream) return;
230
231 UCRA_StreamState* state = (UCRA_StreamState*)stream;
232
233 /* Lock mutex to ensure thread safety */
234 pthread_mutex_lock(&state->mutex);
235
236 /* Mark as closed */
237 state->is_closed = 1;
238
239 /* Wake up any waiting readers */
240 pthread_cond_broadcast(&state->data_available);
241
242 pthread_mutex_unlock(&state->mutex);
243
244 /* Cleanup resources */
245 pthread_cond_destroy(&state->data_available);
246 pthread_mutex_destroy(&state->mutex);
247
248 free(state->buffer);
249 free(state);
250}

252/* Private function to refill the stream buffer by calling the user callback */
253static UCRA_Result refill_stream_buffer(UCRA_StreamState* state) {
254 /* This function should be called with the mutex locked */
255
256 if (state->is_closed) {
257 return UCRA_ERR_INTERNAL;
258 }
259
260 /* Check if we have space for at least one block */
261 uint32_t available_space = get_available_space_frames(state);
262 if (available_space < state->config.block_size) {
263 /* Buffer is full, nothing to do */
264 return UCRA_SUCCESS;
265 }
266
267 /* Prepare render config for callback */
268 UCRA_RenderConfig render_config = state->config;
269
270 /* Call user callback to get next render configuration */
271 UCRA_Result callback_result = state->callback(state->user_data, &render_config);
272 if (callback_result != UCRA_SUCCESS) {
273 return callback_result;
274 }
275
276 /* Fill buffer more aggressively - render multiple blocks if space available */
277 uint32_t blocks_to_render = available_space / state->config.block_size;
278 if (blocks_to_render == 0) {
279 blocks_to_render = 1; /* Always render at least one block if we have any space */
280 }
281 /* Limit to reasonable number to avoid long blocking */
282 if (blocks_to_render > 4) {
283 blocks_to_render = 4;
284 }
285
286 uint32_t frames_to_write = blocks_to_render * state->config.block_size;
287 uint32_t channels = state->config.channels;
288
289 /* Ensure we don't overflow the buffer */
290 if (frames_to_write > available_space) {
291 frames_to_write = available_space;
292 }
293
294 /* Allocate temporary buffer for rendering */
295 float* temp_buffer = malloc(frames_to_write * channels * sizeof(float));
296 if (!temp_buffer) {
297 return UCRA_ERR_OUT_OF_MEMORY;
298 }
299
300 /* Render audio using the provided configuration */
301 UCRA_Result render_result = render_audio_from_notes(state, &render_config,
302 temp_buffer, frames_to_write);
303 if (render_result != UCRA_SUCCESS) {
304 free(temp_buffer);
305 return render_result;
306 }
307
308 /* Copy rendered data to ring buffer, handling wraparound */
309 uint32_t frames_copied = 0;
310 while (frames_copied < frames_to_write) {
311 uint32_t frames_until_wrap = state->buffer_size_frames - state->write_pos;
312 uint32_t frames_to_copy = frames_to_write - frames_copied;
313
314 if (frames_to_copy > frames_until_wrap) {
315 frames_to_copy = frames_until_wrap;
316 }
317
318 /* Copy data */
319 size_t src_offset = frames_copied * channels;
320 size_t dst_offset = state->write_pos * channels;
321 size_t samples_to_copy = frames_to_copy * channels;
322
323 memcpy(state->buffer + dst_offset, temp_buffer + src_offset,
324 samples_to_copy * sizeof(float));
325
326 /* Update positions */
327 state->write_pos = advance_position(state->write_pos, frames_to_copy, state->buffer_size_frames);
328 frames_copied += frames_to_copy;
329 }
330
331 /* Update buffer state */
332 state->available_frames += frames_to_write;
333 state->total_frames_generated += frames_to_write;
334
335 /* Update phase for continuous audio generation */
336 double phase_increment = 2.0 * M_PI * 440.0 / state->config.sample_rate; /* A4 frequency for now */
337 state->phase += phase_increment * frames_to_write;
338 if (state->phase >= 2.0 * M_PI) {
339 state->phase = fmod(state->phase, 2.0 * M_PI);
340 }
341
342 free(temp_buffer);
343
344 /* Signal that data is available */
345 pthread_cond_signal(&state->data_available);
346
347 return UCRA_SUCCESS;
348}

350UCRA_Result ucra_stream_read(UCRA_StreamHandle stream,
351 float* out_buffer,
352 uint32_t frame_count,
353 uint32_t* out_frames_read) {
354 if (!stream || !out_buffer || !out_frames_read) {
355 return UCRA_ERR_INVALID_ARGUMENT;
356 }
357
358 UCRA_StreamState* state = (UCRA_StreamState*)stream;
359 *out_frames_read = 0;
360
361 if (!state->is_initialized || state->is_closed) {
362 return UCRA_ERR_INTERNAL;
363 }
364
365 pthread_mutex_lock(&state->mutex);
366
367 uint32_t frames_copied = 0;
368 uint32_t channels = state->config.channels;
369
370 while (frames_copied < frame_count && !state->is_closed) {
371 /* If buffer is running low, try to refill it proactively */
372 if (state->available_frames < state->config.block_size * 2) {
373 UCRA_Result refill_result = refill_stream_buffer(state);
374 if (refill_result != UCRA_SUCCESS) {
375 pthread_mutex_unlock(&state->mutex);
376 return refill_result;
377 }
378 }
379
380 /* If no data available at all, wait for it but with timeout */
381 if (state->available_frames == 0 && !state->is_closed) {
382 /* Try one more refill attempt before waiting */
383 UCRA_Result refill_result = refill_stream_buffer(state);
384 if (refill_result == UCRA_SUCCESS && state->available_frames > 0) {
385 /* Successfully refilled, continue */
386 continue;
387 }
388
389 /* If still no data, break instead of waiting indefinitely */
390 break;
391 }
392
393 /* Copy available data */
394 uint32_t frames_to_copy = frame_count - frames_copied;
395 if (frames_to_copy > state->available_frames) {
396 frames_to_copy = state->available_frames;
397 }
398
399 /* Handle ring buffer wraparound */
400 uint32_t frames_until_wrap = state->buffer_size_frames - state->read_pos;
401 if (frames_to_copy > frames_until_wrap) {
402 frames_to_copy = frames_until_wrap;
403 }
404
405 /* Copy the data */
406 size_t samples_to_copy = frames_to_copy * channels;
407 size_t src_offset = state->read_pos * channels;
408 size_t dst_offset = frames_copied * channels;
409
410 memcpy(out_buffer + dst_offset, state->buffer + src_offset,
411 samples_to_copy * sizeof(float));
412
413 /* Update buffer state */
414 state->read_pos = advance_position(state->read_pos, frames_to_copy, state->buffer_size_frames);
415 state->available_frames -= frames_to_copy;
416 frames_copied += frames_to_copy;
417 }
418
419 pthread_mutex_unlock(&state->mutex);
420
421 *out_frames_read = frames_copied;
422 return UCRA_SUCCESS;
423}
