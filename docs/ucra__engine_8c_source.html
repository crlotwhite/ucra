2 * UCRA Reference Engine (pure C, no WORLD dependency)

26static double midi_to_hz(int16_t midi_note) {
27 if (midi_note < 0) return 0.0; /* unvoiced */
28 return 440.0 * pow(2.0, ((double)midi_note - 69.0) / 12.0);
29}

31/* sample a step-wise curve by picking the last point <= t (seconds). */
32static double sample_f0_curve(const UCRA_F0Curve* c, double t, double fallback_hz) {
33 if (!c || c->length == 0 || !c->time_sec || !c->f0_hz) return fallback_hz;
34 uint32_t idx = 0;
35 for (uint32_t i = 0; i + 1 < c->length; ++i) {
36 if ((double)c->time_sec[i] <= t) idx = i; else break;
37 }
38 return (double)c->f0_hz[idx];
39}
40
41/* sample an envelope curve similarly (step-wise hold). */
42static double sample_env_curve(const UCRA_EnvCurve* c, double t, double fallback) {
43 if (!c || c->length == 0 || !c->time_sec || !c->value) return fallback;
44 uint32_t idx = 0;
45 for (uint32_t i = 0; i + 1 < c->length; ++i) {
46 if ((double)c->time_sec[i] <= t) idx = i; else break;
47 }
48 return (double)c->value[idx];
49}

51UCRA_Result ucra_engine_create(UCRA_Handle* outEngine,
52 const UCRA_KeyValue* options,
53 uint32_t option_count) {
54 (void)options; (void)option_count;
55 if (!outEngine) return UCRA_ERR_INVALID_ARGUMENT;
56 UCRA_Engine_* eng = (UCRA_Engine_*)calloc(1, sizeof(UCRA_Engine_));
57 if (!eng) return UCRA_ERR_OUT_OF_MEMORY;
58 eng->sample_rate = 44100.0; /* default */

83UCRA_Result ucra_render(UCRA_Handle engine,
84 const UCRA_RenderConfig* config,
85 UCRA_RenderResult* outResult) {
86 UCRA_Engine_* eng = (UCRA_Engine_*)engine;
87 if (!eng || !config || !outResult) return UCRA_ERR_INVALID_ARGUMENT;
88
89 /* Adopt sample rate from config if provided */
90 if (config->sample_rate > 0) eng->sample_rate = (double)config->sample_rate;
91 uint32_t channels = config->channels > 0 ? config->channels : 1;
92
93 /* compute total duration from notes */
94 double total_dur = 0.0;
95 for (uint32_t i = 0; i < config->note_count; ++i) {
96 double end = config->notes[i].start_sec + config->notes[i].duration_sec;
97 if (end > total_dur) total_dur = end;
98 }
99
100 if (total_dur <= 0.0) {
101 outResult->pcm = NULL;
102 outResult->frames = 0;
103 outResult->channels = channels;
104 outResult->sample_rate = (uint32_t)eng->sample_rate;
105 outResult->metadata = NULL;
106 outResult->metadata_count = 0;
107 outResult->status = UCRA_SUCCESS;
108 return UCRA_SUCCESS;
109 }
110
111 uint64_t frames = (uint64_t)(total_dur * eng->sample_rate + 0.5);
112 if (frames == 0) frames = 1;
113 size_t total_samples = (size_t)frames * (size_t)channels;
114
115 /* (re)allocate engine-owned buffer */
116 if (eng->last_pcm) { free(eng->last_pcm); eng->last_pcm = NULL; }
117 eng->last_pcm = (float*)malloc(total_samples * sizeof(float));
118 if (!eng->last_pcm) {
119 outResult->status = UCRA_ERR_OUT_OF_MEMORY;
120 return UCRA_ERR_OUT_OF_MEMORY;
121 }
122 eng->last_pcm_size = total_samples;
123
124 /* zero initialize */
125 for (size_t i = 0; i < total_samples; ++i) eng->last_pcm[i] = 0.0f;
126
127 /* naive additive synthesis: sine at per-note F0, scaled by velocity and optional env */
128 for (uint64_t n = 0; n < frames; ++n) {
129 double t = (double)n / eng->sample_rate;
130 double mix = 0.0;
131 for (uint32_t i = 0; i < config->note_count; ++i) {
132 const UCRA_NoteSegment* note = &config->notes[i];
133 double start = note->start_sec;
134 double end = note->start_sec + note->duration_sec;
135 if (t < start || t > end) continue;
136
137 double rel_t = t - start;
138 double f0 = note->f0_override ? sample_f0_curve(note->f0_override, rel_t, midi_to_hz(note->midi_note))
139 : midi_to_hz(note->midi_note);
140 if (f0 <= 0.0) continue;
141 double env = note->env_override ? sample_env_curve(note->env_override, rel_t, 1.0) : 1.0;
142 double vel = (double)(note->velocity) / 127.0; /* 0..1 */
143 double amp = 0.2 * vel * env; /* conservative gain to avoid clipping */
144 mix += amp * sin(2.0 * M_PI * f0 * t);
145 }
146 /* simple soft clip */
147 if (mix > 1.0) mix = 1.0; else if (mix < -1.0) mix = -1.0;
148 float sample = (float)mix;
149 for (uint32_t ch = 0; ch < channels; ++ch) {
150 eng->last_pcm[n * channels + ch] = sample;
151 }
152 }
153
154 outResult->pcm = eng->last_pcm;
155 outResult->frames = frames;
156 outResult->channels = channels;
157 outResult->sample_rate = (uint32_t)eng->sample_rate;
158 outResult->metadata = NULL;
159 outResult->metadata_count = 0;
160 outResult->status = UCRA_SUCCESS;
161 return UCRA_SUCCESS;
162}
