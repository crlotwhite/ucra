18static int ucra_find_arg(int argc, char* argv[], const char* short_opt, const char* long_opt, char** value) {
19 *value = NULL;
20
21 for (int i = 1; i < argc; i++) {
22 if (strcmp(argv[i], short_opt) == 0 || strcmp(argv[i], long_opt) == 0) {
23 if (i + 1 < argc && argv[i + 1][0] != '-') {
24 *value = argv[i + 1];
25 return i;
26 } else {
27 return i; /* Found flag but no value */
28 }
29 }
30 }
31 return 0; /* Not found */
32}

89/* Parse note information string */
90static UCRA_Result ucra_parse_note_info(const char* note_str, UCRA_CLIArgs* args) {
91 if (!note_str || !args) {
92 return UCRA_ERR_INVALID_ARGUMENT;
93 }
94
95 /* UTAU note format: "lyric note velocity" or more complex formats */
96 char* note_copy = strdup(note_str);
97 if (!note_copy) {
98 return UCRA_ERR_OUT_OF_MEMORY;
99 }
100
101 char* token;
102 int token_count = 0;
103
104 /* Simple parsing: space-separated values */
105 token = strtok(note_copy, " \t");
106 while (token && token_count < 3) {
107 switch (token_count) {
108 case 0: /* lyric */
109 args->lyric = strdup(token);
110 break;
111 case 1: /* MIDI note */
112 args->midi_note = (int16_t)atoi(token);
113 if (args->midi_note < 0 || args->midi_note > 127) {
114 args->midi_note = 60; /* default to C4 */
115 }
116 break;
117 case 2: /* velocity */
118 args->velocity = (uint8_t)atoi(token);
119 if (args->velocity > 127) {
120 args->velocity = 100; /* default velocity */
121 }
122 break;
123 }
124 token = strtok(NULL, " \t");
125 token_count++;
126 }
127
128 free(note_copy);
129
130 /* If no lyric was provided, use default */
131 if (!args->lyric) {
132 args->lyric = strdup("a");
133 }
134
135 return UCRA_SUCCESS;
136}

158/* Parse command line arguments using cross-platform approach */
159static UCRA_Result ucra_parse_cli_args(int argc, char* argv[], UCRA_CLIArgs* args) {
160 if (!args) {
161 return UCRA_ERR_INVALID_ARGUMENT;
162 }
163
164 /* Check for help first */
165 if (ucra_has_flag(argc, argv, "-h", "--help")) {
166 ucra_print_usage(argv[0]);
167 return UCRA_ERR_INVALID_ARGUMENT; /* Use as "help requested" signal */
168 }
169
170 char* value;
171
172 /* Parse required arguments */
173 if (ucra_find_arg(argc, argv, "-i", "--input", &value) && value) {
174 args->input_wav = strdup(value);
175 }
176
177 if (ucra_find_arg(argc, argv, "-o", "--output", &value) && value) {
178 args->output_wav = strdup(value);
179 }
180
181 if (ucra_find_arg(argc, argv, "-n", "--note", &value) && value) {
182 args->note_info = strdup(value);
183 }
184
185 if (ucra_find_arg(argc, argv, "-v", "--vb-root", &value) && value) {
186 args->vb_root = strdup(value);
187 }
188
189 /* Parse optional arguments */
190 if (ucra_find_arg(argc, argv, "-t", "--tempo", &value) && value) {
191 args->tempo = atof(value);
192 if (args->tempo <= 0.0) {
193 args->tempo = 120.0;
194 }
195 }
196
197 if (ucra_find_arg(argc, argv, "-f", "--flags", &value) && value) {
198 args->flags_str = strdup(value);
199 }
200
201 if (ucra_find_arg(argc, argv, "-c", "--f0-curve", &value) && value) {
202 args->f0_curve_file = strdup(value);
203 }
204
205 if (ucra_find_arg(argc, argv, "-O", "--oto", &value) && value) {
206 args->oto_file = strdup(value);
207 }
208
209 if (ucra_find_arg(argc, argv, "-r", "--rate", &value) && value) {
210 args->sample_rate = (uint32_t)atoi(value);
211 if (args->sample_rate < 8000 || args->sample_rate > 192000) {
212 args->sample_rate = 44100;
213 }
214 }
215
216 /* Validate required arguments */
217 if (!args->input_wav) {
218 fprintf(stderr, "Error: Input WAV file is required (--input)\n");
219 return UCRA_ERR_INVALID_ARGUMENT;
220 }
221
222 if (!args->output_wav) {
223 fprintf(stderr, "Error: Output WAV file is required (--output)\n");
224 return UCRA_ERR_INVALID_ARGUMENT;
225 }
226
227 if (!args->note_info) {
228 fprintf(stderr, "Error: Note information is required (--note)\n");
229 return UCRA_ERR_INVALID_ARGUMENT;
230 }
231
232 if (!args->vb_root) {
233 fprintf(stderr, "Error: Voicebank root directory is required (--vb-root)\n");
234 return UCRA_ERR_INVALID_ARGUMENT;
235 }
236
237 /* Parse note information */
238 UCRA_Result result = ucra_parse_note_info(args->note_info, args);
239 if (result != UCRA_SUCCESS) {
240 fprintf(stderr, "Error: Failed to parse note information\n");
241 return result;
242 }
243
244 return UCRA_SUCCESS;
245}

272/* Load F0 curve from file */
273static UCRA_Result ucra_load_f0_curve(const char* f0_file, UCRA_F0Curve* curve) {
274 if (!f0_file || !curve) {
275 return UCRA_ERR_INVALID_ARGUMENT;
276 }
277
278 FILE* file = fopen(f0_file, "r");
279 if (!file) {
280 return UCRA_ERR_FILE_NOT_FOUND;
281 }
282
283 /* Count lines first */
284 uint32_t line_count = 0;
285 char buffer[256];
286 while (fgets(buffer, sizeof(buffer), file)) {
287 line_count++;
288 }
289
290 if (line_count == 0) {
291 fclose(file);
292 return UCRA_ERR_INVALID_ARGUMENT;
293 }
294
295 /* Allocate arrays */
296 float* time_sec = malloc(line_count * sizeof(float));
297 float* f0_hz = malloc(line_count * sizeof(float));
298 if (!time_sec || !f0_hz) {
299 free(time_sec);
300 free(f0_hz);
301 fclose(file);
302 return UCRA_ERR_OUT_OF_MEMORY;
303 }
304
305 /* Read data */
306 rewind(file);
307 uint32_t valid_count = 0;
308 while (fgets(buffer, sizeof(buffer), file) && valid_count < line_count) {
309 float time, f0;
310 if (sscanf(buffer, "%f %f", &time, &f0) == 2) {
311 time_sec[valid_count] = time;
312 f0_hz[valid_count] = f0;
313 valid_count++;
314 }
315 }
316
317 fclose(file);
318
319 if (valid_count == 0) {
320 free(time_sec);
321 free(f0_hz);
322 return UCRA_ERR_INVALID_ARGUMENT;
323 }
324
325 curve->time_sec = time_sec;
326 curve->f0_hz = f0_hz;
327 curve->length = valid_count;
328
329 return UCRA_SUCCESS;
330}

359/* Convert CLI args to UCRA_RenderConfig */
360static UCRA_Result ucra_cli_to_render_config(const UCRA_CLIArgs* args, const UCRA_NoteSegment* note,
361 UCRA_RenderConfig* config, UCRA_KeyValue* options) {
362 if (!args || !note || !config) {
363 return UCRA_ERR_INVALID_ARGUMENT;
364 }
365
366 memset(config, 0, sizeof(UCRA_RenderConfig));
367
368 config->sample_rate = args->sample_rate;
369 config->channels = 1; /* Mono output for compatibility */
370 config->block_size = 512; /* Default block size */
371 config->flags = 0;
372
373 config->notes = note;
374 config->note_count = 1;
375
376 /* Apply flag mapping if provided */
377 if (args->flags_str && options) {
378 /* Parse legacy flags */
379 UCRA_KeyValue* legacy_flags = NULL;
380 uint32_t legacy_count = 0;
381 UCRA_Result result = ucra_parse_legacy_flags(args->flags_str, &legacy_flags, &legacy_count);
382
383 if (result == UCRA_SUCCESS && legacy_count > 0) {
384 /* Try to load default moresampler mapping */
385 UCRA_FlagMapper* mapper = NULL;
386 const char* mapping_path = "tools/flag_mapper/mappings/moresampler_map.json";
387
388 if (ucra_flag_mapper_load(mapping_path, &mapper) == UCRA_SUCCESS) {
389 UCRA_FlagMapResult map_result;
390 if (ucra_flag_mapper_apply(mapper, legacy_flags, legacy_count, &map_result) == UCRA_SUCCESS) {
391 /* Use mapped flags */
392 config->options = map_result.flags;
393 config->option_count = map_result.flag_count;
394
395 /* Print warnings if any */
396 for (uint32_t i = 0; i < map_result.warning_count; i++) {
397 fprintf(stderr, "Flag mapping warning: %s\n", map_result.warnings[i]);
398 }
399
400 /* Note: Caller is responsible for cleanup of map_result */
401 } else {
402 printf("Warning: Flag mapping failed, using raw flags\n");
403 /* Fallback to simple key=value parsing */
404 config->options = options;
405 config->option_count = 1;
406 options[0].key = "flags";
407 options[0].value = args->flags_str;
408 }
409 ucra_flag_mapper_free(mapper);
410 } else {
411 printf("Warning: Could not load flag mapping, using raw flags\n");
412 /* Fallback to simple key=value parsing */
413 config->options = options;
414 config->option_count = 1;
415 options[0].key = "flags";
416 options[0].value = args->flags_str;
417 }
418
419 ucra_free_legacy_flags(legacy_flags, legacy_count);
420 } else {
421 /* Fallback to simple key=value parsing */
422 config->options = options;
423 config->option_count = 1;
424 options[0].key = "flags";
425 options[0].value = args->flags_str;
426 }
427 }
428
429 return UCRA_SUCCESS;
430}

432/* Write PCM data to WAV file (simplified implementation) */
433static UCRA_Result ucra_write_wav_file(const char* filename, const float* pcm, uint64_t frames,
434 uint32_t channels, uint32_t sample_rate) {
435 if (!filename || !pcm || frames == 0) {
436 return UCRA_ERR_INVALID_ARGUMENT;
437 }
438
439 FILE* file = fopen(filename, "wb");
440 if (!file) {
441 return UCRA_ERR_FILE_NOT_FOUND;
442 }
443
444 /* Write basic WAV header (simplified) */
445 uint32_t data_size = (uint32_t)(frames * channels * sizeof(float));
446 uint32_t file_size = data_size + 36;
447
448 /* RIFF header */
449 fwrite("RIFF", 1, 4, file);
450 fwrite(&file_size, 4, 1, file);
451 fwrite("WAVE", 1, 4, file);
452
453 /* fmt chunk */
454 fwrite("fmt ", 1, 4, file);
455 uint32_t fmt_size = 16;
456 uint16_t audio_format = 3; /* IEEE float */
457 uint16_t num_channels = (uint16_t)channels;
458 uint32_t byte_rate = sample_rate * channels * sizeof(float);
459 uint16_t block_align = (uint16_t)(channels * sizeof(float));
460 uint16_t bits_per_sample = 32;
461
462 fwrite(&fmt_size, 4, 1, file);
463 fwrite(&audio_format, 2, 1, file);
464 fwrite(&num_channels, 2, 1, file);
465 fwrite(&sample_rate, 4, 1, file);
466 fwrite(&byte_rate, 4, 1, file);
467 fwrite(&block_align, 2, 1, file);
468 fwrite(&bits_per_sample, 2, 1, file);
469
470 /* data chunk */
471 fwrite("data", 1, 4, file);
472 fwrite(&data_size, 4, 1, file);
473 fwrite(pcm, sizeof(float), frames * channels, file);
474
475 fclose(file);
476 return UCRA_SUCCESS;
477}
