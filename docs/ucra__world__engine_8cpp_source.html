57/* Helper function to convert MIDI note to frequency */
58static double midi_note_to_frequency(int16_t midi_note) {
59 if (midi_note < 0) {
60 return 440.0; /* Default to A4 */
61 }
62 return 440.0 * std::pow(2.0, (midi_note - 69) / 12.0);
63}

65/* Convert UCRA note segments to WORLD-compatible F0 and time arrays */
66static void prepare_world_f0_data(const UCRA_NoteSegment* notes, uint32_t note_count,
67 double sample_rate, double frame_period_ms,
68 double total_duration_sec,
69 std::vector<double>& f0_array,
70 std::vector<double>& time_array) {
71 /* Calculate number of frames */
72 int frame_count = static_cast<int>(1000.0 * total_duration_sec / frame_period_ms) + 1;
73
74 f0_array.resize(frame_count, 0.0);
75 time_array.resize(frame_count);
76
77 /* Fill time array */
78 for (int i = 0; i < frame_count; i++) {
79 time_array[i] = i * frame_period_ms / 1000.0;
80 }
81
82 /* Fill F0 array based on note segments */
83 for (uint32_t note_idx = 0; note_idx < note_count; note_idx++) {
84 const UCRA_NoteSegment* note = &notes[note_idx];
85
86 if (note->midi_note < 0) continue; /* Skip non-pitched notes */
87
88 double note_f0 = midi_note_to_frequency(note->midi_note);
89
90 /* Find frames that correspond to this note */
91 int start_frame = static_cast<int>(note->start_sec * 1000.0 / frame_period_ms);
92 int end_frame = static_cast<int>((note->start_sec + note->duration_sec) * 1000.0 / frame_period_ms);
93
94 /* Clamp to valid range */
95 start_frame = std::max(0, std::min(start_frame, frame_count - 1));
96 end_frame = std::max(0, std::min(end_frame, frame_count - 1));
97
98 /* Fill F0 for this note */
99 for (int frame = start_frame; frame <= end_frame; frame++) {
100 if (note->f0_override && note->f0_override->length > 0) {
101 /* Use F0 override if available */
102 double relative_time = time_array[frame] - note->start_sec;
103 /* Simple linear interpolation for F0 override */
104 if (relative_time >= 0 && relative_time <= note->duration_sec) {
105 /* Find closest point in F0 curve */
106 uint32_t curve_idx = 0;
107 for (uint32_t i = 0; i < note->f0_override->length - 1; i++) {
108 if (note->f0_override->time_sec[i] <= relative_time) {
109 curve_idx = i;
110 }
111 }
112 f0_array[frame] = note->f0_override->f0_hz[curve_idx];
113 }
114 } else {
115 /* Use MIDI note frequency */
116 f0_array[frame] = note_f0;
117 }
118 }
119 }
120}

122UCRA_Result ucra_engine_create(UCRA_Handle* outEngine,
123 const UCRA_KeyValue* options,
124 uint32_t option_count) {
125 if (!outEngine) {
126 return UCRA_ERR_INVALID_ARGUMENT;
127 }
128
129 /* For simplicity, create a single global engine instance */
130 if (g_world_engine != nullptr) {
131 *outEngine = reinterpret_cast<UCRA_Handle>(g_world_engine);
132 return UCRA_SUCCESS;
133 }
134
135 /* Allocate engine state */
136 g_world_engine = static_cast<UCRA_WorldEngine*>(malloc(sizeof(UCRA_WorldEngine)));
137 if (!g_world_engine) {
138 return UCRA_ERR_OUT_OF_MEMORY;
139 }
140
141 /* Initialize with default parameters */
142 g_world_engine->sample_rate = 44100.0;
143 g_world_engine->frame_period = 5.0; /* 5ms frame period */
144
145 /* Initialize WORLD analysis options */
146 InitializeDioOption(&g_world_engine->dio_option);
147 InitializeHarvestOption(&g_world_engine->harvest_option);
148 InitializeCheapTrickOption(static_cast<int>(g_world_engine->sample_rate), &g_world_engine->cheaptrick_option);
149 InitializeD4COption(&g_world_engine->d4c_option);
150
151 /* Get FFT size from CheapTrick option */
152 g_world_engine->fft_size = GetFFTSizeForCheapTrick(static_cast<int>(g_world_engine->sample_rate), &g_world_engine->cheaptrick_option);
153 g_world_engine->synthesis_frame_period = static_cast<int>(g_world_engine->frame_period);
154
155 /* Set frame period for DIO and Harvest options */
156 g_world_engine->dio_option.frame_period = g_world_engine->frame_period;
157 g_world_engine->harvest_option.frame_period = g_world_engine->frame_period;
158
159 /* Initialize memory for results */
160 g_world_engine->last_pcm = nullptr;
161 g_world_engine->last_pcm_size = 0;
162 g_world_engine->last_metadata = nullptr;
163 g_world_engine->last_metadata_count = 0;
164
165 /* Process options if provided */
166 for (uint32_t i = 0; i < option_count; i++) {
167 const char* key = options[i].key;
168 const char* value = options[i].value;
169
170 if (key && value) {
171 if (strcmp(key, "sample_rate") == 0) {
172 double sr = atof(value);
173 if (sr > 0) {
174 g_world_engine->sample_rate = sr;
175 InitializeCheapTrickOption(static_cast<int>(sr), &g_world_engine->cheaptrick_option);
176 g_world_engine->fft_size = GetFFTSizeForCheapTrick(static_cast<int>(sr), &g_world_engine->cheaptrick_option);
177 }
178 } else if (strcmp(key, "frame_period") == 0) {
179 double fp = atof(value);
180 if (fp > 0) {
181 g_world_engine->frame_period = fp;
182 g_world_engine->synthesis_frame_period = static_cast<int>(fp);
183 g_world_engine->dio_option.frame_period = fp;
184 g_world_engine->harvest_option.frame_period = fp;
185 }
186 }
187 }
188 }
189
190 /* Prepare engine info string */
191 snprintf(g_world_engine->engine_info, sizeof(g_world_engine->engine_info),
192 "WORLD Vocoder Engine v1.0 (sample_rate=%.1f, frame_period=%.1f)",
193 g_world_engine->sample_rate, g_world_engine->frame_period);
194
195 *outEngine = reinterpret_cast<UCRA_Handle>(g_world_engine);
196 return UCRA_SUCCESS;
197}

199void ucra_engine_destroy(UCRA_Handle engine) {
200 if (!engine || engine != reinterpret_cast<UCRA_Handle>(g_world_engine)) {
201 return;
202 }
203
204 UCRA_WorldEngine* world_engine = reinterpret_cast<UCRA_WorldEngine*>(engine);
205
206 /* Free last render results */
207 if (world_engine->last_pcm) {
208 free(world_engine->last_pcm);
209 world_engine->last_pcm = nullptr;
210 }
211
212 if (world_engine->last_metadata) {
213 free(world_engine->last_metadata);
214 world_engine->last_metadata = nullptr;
215 }
216
217 /* Free engine state */
218 free(world_engine);
219 g_world_engine = nullptr;
220}

240UCRA_Result ucra_render(UCRA_Handle engine,
241 const UCRA_RenderConfig* config,
242 UCRA_RenderResult* outResult) {
243 if (!engine || !config || !outResult) {
244 return UCRA_ERR_INVALID_ARGUMENT;
245 }
246
247 UCRA_WorldEngine* world_engine = reinterpret_cast<UCRA_WorldEngine*>(engine);
248
249 /* Update engine parameters from config */
250 if (config->sample_rate > 0 && config->sample_rate != world_engine->sample_rate) {
251 world_engine->sample_rate = config->sample_rate;
252 InitializeCheapTrickOption(static_cast<int>(config->sample_rate), &world_engine->cheaptrick_option);
253 world_engine->fft_size = GetFFTSizeForCheapTrick(static_cast<int>(config->sample_rate), &world_engine->cheaptrick_option);
254 }
255
256 /* Calculate total duration from notes */
257 double total_duration = 0.0;
258 for (uint32_t i = 0; i < config->note_count; i++) {
259 double note_end = config->notes[i].start_sec + config->notes[i].duration_sec;
260 if (note_end > total_duration) {
261 total_duration = note_end;
262 }
263 }
264
265 if (total_duration <= 0.0) {
266 /* No notes to render - return silence */
267 outResult->pcm = nullptr;
268 outResult->frames = 0;
269 outResult->channels = config->channels;
270 outResult->sample_rate = static_cast<uint32_t>(world_engine->sample_rate);
271 outResult->metadata = nullptr;
272 outResult->metadata_count = 0;
273 outResult->status = UCRA_SUCCESS;
274 return UCRA_SUCCESS;
275 }
276
277 /* Prepare F0 data for WORLD */
278 std::vector<double> f0_array, time_array;
279 prepare_world_f0_data(config->notes, config->note_count,
280 world_engine->sample_rate, world_engine->frame_period,
281 total_duration, f0_array, time_array);
282
283 int frame_count = static_cast<int>(f0_array.size());
284 if (frame_count <= 0) {
285 outResult->status = UCRA_ERR_INTERNAL;
286 return UCRA_ERR_INTERNAL;
287 }
288
289 /* Allocate arrays for spectral envelope and aperiodicity */
290 double** spectrogram = nullptr;
291 double** aperiodicity = nullptr;
292
293 try {
294 /* Allocate 2D arrays for WORLD */
295 spectrogram = new double*[frame_count];
296 aperiodicity = new double*[frame_count];
297
298 for (int i = 0; i < frame_count; i++) {
299 spectrogram[i] = new double[world_engine->fft_size / 2 + 1];
300 aperiodicity[i] = new double[world_engine->fft_size / 2 + 1];
301
302 /* Initialize with default values */
303 for (int j = 0; j < world_engine->fft_size / 2 + 1; j++) {
304 spectrogram[i][j] = -60.0; /* -60dB default */
305 aperiodicity[i][j] = 0.0; /* Voiced by default */
306 }
307 }
308
309 /* Perform CheapTrick analysis (spectral envelope) */
310 /* For synthesis, we'll use a simple spectral model */
311 for (int i = 0; i < frame_count; i++) {
312 if (f0_array[i] > 0.0) {
313 /* Generate harmonic spectrum for voiced frames */
314 double fundamental = f0_array[i];
315 for (int j = 0; j < world_engine->fft_size / 2 + 1; j++) {
316 double freq = static_cast<double>(j) * world_engine->sample_rate / world_engine->fft_size;
317
318 /* Simple harmonic model */
319 if (freq > 0) {
320 double harmonic_num = freq / fundamental;
321 if (harmonic_num >= 1.0 && harmonic_num <= 20.0) {
322 /* Calculate harmonic amplitude (decreasing with harmonic number) */
323 double amplitude = -20.0 * log10(harmonic_num + 1.0);
324 spectrogram[i][j] = amplitude;
325 }
326 }
327 }
328
329 /* Set aperiodicity for voiced frames */
330 for (int j = 0; j < world_engine->fft_size / 2 + 1; j++) {
331 aperiodicity[i][j] = 0.1; /* Slightly aperiodic */
332 }
333 } else {
334 /* Unvoiced frame - set high aperiodicity */
335 for (int j = 0; j < world_engine->fft_size / 2 + 1; j++) {
336 spectrogram[i][j] = -40.0; /* Noise spectrum */
337 aperiodicity[i][j] = 0.9; /* Highly aperiodic */
338 }
339 }
340 }
341
342 /* Synthesize audio using WORLD */
343 int output_length = static_cast<int>(total_duration * world_engine->sample_rate);
344 std::vector<double> synthesized_audio(output_length);
345
346 Synthesis(f0_array.data(), frame_count, spectrogram, aperiodicity,
347 world_engine->fft_size, world_engine->frame_period,
348 static_cast<int>(world_engine->sample_rate), output_length,
349 synthesized_audio.data());
350
351 /* Convert to float and store in engine */
352 size_t total_samples = output_length * config->channels;
353
354 /* Free previous results */
355 if (world_engine->last_pcm) {
356 free(world_engine->last_pcm);
357 }
358
359 world_engine->last_pcm = static_cast<float*>(malloc(total_samples * sizeof(float)));
360 if (!world_engine->last_pcm) {
361 /* Cleanup and return error */
362 for (int i = 0; i < frame_count; i++) {
363 delete[] spectrogram[i];
364 delete[] aperiodicity[i];
365 }
366 delete[] spectrogram;
367 delete[] aperiodicity;
368
369 outResult->status = UCRA_ERR_OUT_OF_MEMORY;
370 return UCRA_ERR_OUT_OF_MEMORY;
371 }
372
373 world_engine->last_pcm_size = total_samples;
374
375 /* Convert to interleaved float and duplicate for multiple channels */
376 for (int sample = 0; sample < output_length; sample++) {
377 float sample_value = static_cast<float>(synthesized_audio[sample]);
378 for (uint32_t ch = 0; ch < config->channels; ch++) {
379 world_engine->last_pcm[sample * config->channels + ch] = sample_value;
380 }
381 }
382
383 /* Fill output result */
384 outResult->pcm = world_engine->last_pcm;
385 outResult->frames = output_length;
386 outResult->channels = config->channels;
387 outResult->sample_rate = static_cast<uint32_t>(world_engine->sample_rate);
388 outResult->metadata = nullptr;
389 outResult->metadata_count = 0;
390 outResult->status = UCRA_SUCCESS;
391
392 /* Cleanup WORLD arrays */
393 for (int i = 0; i < frame_count; i++) {
394 delete[] spectrogram[i];
395 delete[] aperiodicity[i];
396 }
397 delete[] spectrogram;
398 delete[] aperiodicity;
399
400 return UCRA_SUCCESS;
401
402 } catch (...) {
403 /* Exception occurred - cleanup and return error */
404 if (spectrogram) {
405 for (int i = 0; i < frame_count; i++) {
406 delete[] spectrogram[i];
407 }
408 delete[] spectrogram;
409 }
410
411 if (aperiodicity) {
412 for (int i = 0; i < frame_count; i++) {
413 delete[] aperiodicity[i];
414 }
415 delete[] aperiodicity;
416 }
417
418 outResult->status = UCRA_ERR_INTERNAL;
419 return UCRA_ERR_INTERNAL;
420 }
421}
